# Python program to calculate the first 10,000 terms of the series 1 - 1/3 + 1/5 - 1/7 + ...

# Function to calculate the series
def calculate_series(terms):
    total = 0.0
    for i in range(terms):
        if i % 2 == 0:
            total += 1 / (2 * i + 1)  # Adding the term if i is even  
        else:
            total -= 1 / (2 * i + 1)  # Subtracting the term if i is odd
    return total

# Calculate first 10,000 terms
terms = 10000
result = calculate_series(terms) * 4  # Multiplying total by 4

# Output the result
print(f'The result of the series calculation for the first {terms} terms multiplied by 4 is: {result}')

# Output
The result of the series calculation for the first 10000 terms multiplied by 4 is: 3.1414926535900345
